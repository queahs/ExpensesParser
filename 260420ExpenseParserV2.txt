function accOptions(selected) {
  const blank = `<option value=""${!selected?' selected':''}>— select account —</option>`;
  return blank + ACCOUNTS.map(a => {

function excelDateToISO(val) {
  if (!val) return '';
  if (val instanceof Date) return val.toISOString().slice(0,10);
  if (typeof val === 'number') {
    // Excel serial date
    const d = new Date(Math.round((val - 25569) * 86400 * 1000));
    return d.toISOString().slice(0,10);
  }
  if (typeof val === 'string') {
    // Try parsing various formats
    const clean = val.trim();
    // DD/MM/YYYY
    const m1 = clean.match(/^(\d{1,2})\/(\d{1,2})\/(\d{4})$/);
    if (m1) return `${m1[3]}-${m1[2].padStart(2,'0')}-${m1[1].padStart(2,'0')}`;
    // YYYY-MM-DD
    if (/^\d{4}-\d{2}-\d{2}$/.test(clean)) return clean;
    return clean;
  }
  return '';
}

function parseFile(file) {
  const reader = new FileReader();
  reader.onload = function(e) {
    try {
      const wb = XLSX.read(e.target.result, {type:'array', cellDates:true});
      const wsName = wb.SheetNames.includes('Expenses') ? 'Expenses' : wb.SheetNames[0];
      const ws = wb.Sheets[wsName];
      const data = XLSX.utils.sheet_to_json(ws, {header:1, defval:null, raw:false, dateNF:'yyyy-mm-dd'});

      // Row 0 = title, Row 1 = headers, data from row 2 onward
      // Cols: A=Post? B=TypeOfExpense C=VATRate D=PostType E=Reference F=Date G=Supplier H=Description I=GrossAmount
      let imported = 0;
      rows = [];
      nextId = 0;

      for (let i = 2; i < data.length; i++) {
        const r = data[i];
        if (!r || r.every(v => !v)) continue;
        const gross = parseFloat(String(r[8]||'').replace(/[£,]/g,''));
        if (!gross && !r[7] && !r[6]) continue; // skip fully empty

        const accRaw = String(r[1]||'').trim();
        // Use exactly what the client provided — match to our account list, blank if unrecognised
        let accFull = '';
        if (accRaw) {
          const codeMatch = accRaw.match(/^(\d{3})/);
          if (codeMatch) {
            const found = ACCOUNTS.find(a => a.code === codeMatch[1]);
            accFull = found ? `${found.code} - ${found.name}` : ''; // blank if code not in list
          }
          // if no numeric code prefix at all, leave blank — client left it empty or typed something invalid
        }

        const vatRaw = String(r[2]||'').trim();
        // Use exactly what the client provided — match to our VAT list, blank if unrecognised
        const vatMatch = VAT_RATES.find(v => v.toLowerCase() === vatRaw.toLowerCase());
        const vat = vatMatch || ''; // blank if not a recognised VAT rate

        const dateRaw = r[5];
        let dateISO = '';
        if (dateRaw) {
          // Try numeric serial first
          const num = parseFloat(dateRaw);
          if (!isNaN(num) && num > 40000) {
            const d = new Date(Math.round((num - 25569) * 86400 * 1000));
            dateISO = d.toISOString().slice(0,10);
          } else {
            dateISO = excelDateToISO(dateRaw);
          }
        }

        addRow({
          post: String(r[0]||'YES').trim().toUpperCase() === 'NO' ? 'NO' : 'YES',
          acc: accFull,
          vat: vat,
          ref: String(r[4]||'').trim(),
          date: dateISO,
          supplier: String(r[6]||'').trim(),
          desc: String(r[7]||'').trim(),
          gross: isNaN(gross) ? '' : gross,
          bu: '',
          home: false,
        });
        imported++;
      }

      const blankAcc = rows.filter(r => !r.acc).length;
      const blankVat = rows.filter(r => !r.vat).length;
      let alertMsg = `✓ Imported ${imported} row${imported!==1?'s':''} from "${file.name}".`;
      if (blankAcc || blankVat) {
        alertMsg += ` ${blankAcc+blankVat} field${blankAcc+blankVat!==1?'s need':' needs'} attention — rows with missing account or VAT rate are highlighted.`;
      } else {
        alertMsg += ` Review % BU and home proportion as needed, then export.`;
      }
      setAlert(alertMsg, blankAcc||blankVat ? 'warn' : 'success');
      showTable();
      checkHomeRows();
    } catch(err) {
      setAlert('Could not read file: ' + err.message + '. Please ensure it is the standard client expense template (.xlsx).', 'warn');
    }
  };
  reader.readAsArrayBuffer(file);
}

// ── MSG FILE PARSER ───────────────────────────────────────────────────────────

function parseMsgFile(file) {
  setAlert('Reading email file…', 'info');
  const reader = new FileReader();
  reader.onload = function(e) {
    try {
      // .msg files are OLE Compound Document format (binary)
      // Extract readable text by scanning for UTF-16 and ASCII text runs
      const buf = e.target.result;
      const bytes = new Uint8Array(buf);
      const text = extractMsgText(bytes);

      if (!text || text.trim().length < 10) {
        setAlert('Could not read email content. Try copying the email body and pasting it in the text box below instead.', 'warn');
        return;
      }

      // Populate the paste box and trigger extraction
      document.getElementById('emailPasteBox').value = text;
      setAlert('Email content extracted — running expense detection…', 'info');
      parseEmail();
    } catch(err) {
      setAlert('Could not read .msg file: ' + err.message + '. Try pasting the email text instead.', 'warn');
    }
  };
  reader.readAsArrayBuffer(file);
}

function extractMsgText(bytes) {
  // Scan for readable ASCII/UTF-8 text runs of meaningful length
  // MSG files store body text in multiple formats; we grab the longest clean run
  const decoder = new TextDecoder('utf-8', { fatal: false });
  const decoder16 = new TextDecoder('utf-16le', { fatal: false });

  let best = '';

  // Try UTF-16LE runs (Outlook stores body as UTF-16)
  for (let i = 0; i < bytes.length - 40; i++) {
    // Look for runs where every other byte is 0 (UTF-16LE ASCII range)
    if (bytes[i+1] === 0 && bytes[i] >= 0x20 && bytes[i] < 0x7f) {
      let end = i;
      while (end < bytes.length - 1 && bytes[end+1] === 0 && (bytes[end] >= 0x09)) {
        end += 2;
      }
      const len = end - i;
      if (len > 80) {
        const chunk = decoder16.decode(bytes.slice(i, end));
        const cleaned = chunk.replace(/[\x00-\x08\x0b\x0c\x0e-\x1f]/g, '').trim();
        if (cleaned.length > best.length && /[a-zA-Z]{3,}/.test(cleaned)) {
          best = cleaned;
        }
      }
    }
  }

  // Also try plain UTF-8/ASCII runs
  let start = -1;
  for (let i = 0; i < bytes.length; i++) {
    const b = bytes[i];
    const printable = (b >= 0x20 && b < 0x7f) || b === 0x09 || b === 0x0a || b === 0x0d;
    if (printable) {
      if (start === -1) start = i;
    } else {
      if (start !== -1 && (i - start) > 120) {
        const chunk = decoder.decode(bytes.slice(start, i)).trim();
        const cleaned = stripHtml(chunk);
        if (cleaned.length > best.length && /[a-zA-Z]{3,}/.test(cleaned)) {
          best = cleaned;
        }
      }
      start = -1;
    }
  }

  return best.slice(0, 4000); // cap at 4000 chars — enough for any expense email
}

function stripHtml(html) {
  return html
    .replace(/<style[^>]*>[\s\S]*?<\/style>/gi, '')
    .replace(/<script[^>]*>[\s\S]*?<\/script>/gi, '')
    .replace(/<[^>]+>/g, ' ')
    .replace(/&nbsp;/g, ' ')
    .replace(/&amp;/g, '&')
    .replace(/&lt;/g, '<')
    .replace(/&gt;/g, '>')
    .replace(/\s{2,}/g, ' ')
    .trim();
}



function getQuarterEndDate() {
  const quarter = document.getElementById('quarter').value;
  const year = document.getElementById('yearInput').value || new Date().getFullYear();
  const ends = { Mar: `${year}-03-31`, Jun: `${year}-06-30`, Sep: `${year}-09-30`, Dec: `${year}-12-31` };
  return ends[quarter] || `${year}-03-31`;
}

async function parseEmail() {
  const text = document.getElementById('emailPasteBox').value.trim();
  if (!text) {
    setEmailStatus('Please paste some email text first.', 'warn');
    return;
  }

  const btn = document.getElementById('parseEmailBtn');
  const fallbackDate = getQuarterEndDate();

  btn.disabled = true;
  setEmailStatus('Extracting expenses…', 'info');

  try {
    const systemPrompt = `You are an expense extraction assistant. Extract all expense items from the email text provided.
Return ONLY a valid JSON array — no markdown, no explanation, no code fences.
Each item must have these fields:
  date: ISO date string YYYY-MM-DD if found, otherwise null
  supplier: the company or person paid, or null if unclear
  description: short description of the expense
  amount: numeric amount in GBP (number, not string), or null if not found

Example output:
[{"date":"2025-03-15","supplier":"EE","description":"Mobile phone bill","amount":45.00},{"date":null,"supplier":"Shell","description":"Fuel","amount":60.00}]`;

    const response = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 1000,
        system: systemPrompt,
        messages: [{ role: 'user', content: text }]
      })
    });

    const data = await response.json();
    if (!response.ok) throw new Error(data.error?.message || 'API error');

    const raw = data.content.find(b => b.type === 'text')?.text || '';
    const clean = raw.replace(/```json|```/g, '').trim();
    const expenses = JSON.parse(clean);

    if (!Array.isArray(expenses) || expenses.length === 0) {
      setEmailStatus('No expenses found in that text. Try rephrasing or check the content.', 'warn');
      btn.disabled = false;
      return;
    }

    let added = 0;
    expenses.forEach(e => {
      addRow({
        post: 'YES',
        acc: '',
        vat: '',
        ref: '',
        date: e.date || fallbackDate,
        supplier: e.supplier || '',
        desc: e.description || '',
        gross: e.amount !== null && !isNaN(e.amount) ? e.amount : '',
        bu: '',
        home: false,
      });
      added++;
    });

    document.getElementById('emailPasteBox').value = '';
    showTable();
    checkHomeRows();
    setEmailStatus(`✓ ${added} expense${added!==1?'s':''} added — set account codes and VAT rates for highlighted rows.`, 'success');
    setAlert(`✓ ${added} row${added!==1?'s':''} extracted from email. Account and VAT need filling in.`, 'warn');

  } catch(err) {
    setEmailStatus('Could not extract expenses: ' + err.message, 'warn');
  }

  btn.disabled = false;
}

function setEmailStatus(msg, type) {
  const el = document.getElementById('emailStatus');
  const colours = { info: 'var(--text-muted)', success: 'var(--green)', warn: 'var(--amber)' };
  el.style.color = colours[type] || 'var(--text-muted)';
  el.textContent = msg;
}



function addRow(data) {
  const id = nextId++;
  const d = data || {};
  rows.push({
    id,
    post: d.post || 'YES',
    acc: d.acc || '',
    vat: d.vat || '',
    ref: d.ref || '',
    date: d.date || '',
    supplier: d.supplier || '',
    desc: d.desc || '',
    gross: d.gross !== undefined ? d.gross : '',
    bu: d.bu || '',
    home: d.home || false,
  });
  renderRow(rows[rows.length-1]);
  updateUI();
}

function removeRow(id) {
  rows = rows.filter(r => r.id !== id);
  const el = document.getElementById('tr_'+id);
  if (el) el.remove();
  updateUI();
  if (rows.length === 0) {
    document.getElementById('expTable').style.display = 'none';
    document.getElementById('emptyState').style.display = 'block';
  }
}

function renderRow(row) {
  const tbody = document.getElementById('expBody');
  const tr = document.createElement('tr');
  tr.id = 'tr_' + row.id;
  if (row.post === 'NO') tr.classList.add('excluded');
  if (!row.acc || !row.vat) tr.classList.add('needs-attention');
  tr.innerHTML = buildRowHTML(row);
  tbody.appendChild(tr);
}

  const id = row.id;
  const gross = row.gross !== '' ? parseFloat(row.gross) : 0;

  const netStr = isNaN(net) ? '—' : `£${net.toFixed(2)}`;
  return `
    <td class="no-strike">
      <button class="toggle-post ${row.post==='YES'?'yes':'no'}" id="postBtn_${id}"
        onclick="togglePost(${id})">${row.post}</button>
    </td>
    <td><select onchange="onAccChange(${id},this.value)">${accOptions(row.acc)}</select></td>

function esc(s) {
  return String(s||'').replace(/"/g,'&quot;').replace(/</g,'&lt;');
}

function setField(id, field, val) {
  const row = rows.find(r => r.id === id);
  if (!row) return;
  row[field] = val;
  if (field === 'vat' || field === 'acc') {
    const tr = document.getElementById('tr_'+id);
    if (tr && row.acc && row.vat) tr.classList.remove('needs-attention');
  }
  updateSummary();
  refreshExportBtn();
}

function propagateBU(id, val) {
  const source = rows.find(r => r.id === id);
  if (!source || !source.acc.startsWith('203')) return;
  const supplier = (source.supplier || '').trim().toLowerCase();
  if (!supplier) return;

  rows.forEach(r => {
    if (r.id === id) return;
    if (!r.acc.startsWith('203')) return;
    const rSupplier = (r.supplier || '').trim().toLowerCase();
    // Also read live supplier from DOM in case it was edited
    const supEl = document.querySelector(`#tr_${r.id} td:nth-child(6) input`);
    const domSupplier = supEl ? supEl.value.trim().toLowerCase() : rSupplier;
    if (domSupplier !== supplier) return;

    // Apply same BU
    r.bu = val;
    const buEl = document.querySelector(`#tr_${r.id} .bu-cell input`);
    if (buEl) buEl.value = val;
    updateNet(r.id);
  });
}

function onAccChange(id, val) {
  const row = rows.find(r => r.id === id);
  if (!row) return;
  row.acc = val;
  const dv = defaultVat(val);
  if (dv) {
    row.vat = dv;
  }
  const tr = document.getElementById('tr_'+id);
  if (tr) {
    tr.innerHTML = buildRowHTML(row);
    if (!row.acc || !row.vat) tr.classList.add('needs-attention');
    else tr.classList.remove('needs-attention');
  }
  // Propagate acc+vat to rows with same description
  propagateDescMatch(id, 'acc');
  checkHomeRows();
  updateSummary();
  refreshExportBtn();
}

function onVatChange(id, val) {
  const row = rows.find(r => r.id === id);
  if (!row) return;
  row.vat = val;
  const tr = document.getElementById('tr_'+id);
  if (tr && row.acc && row.vat) tr.classList.remove('needs-attention');
  // Propagate vat to rows with same description
  propagateDescMatch(id, 'vat');
  updateSummary();
  refreshExportBtn();
}

// Track rows that were auto-propagated so user can undo per-row
const propagatedRows = new Set();

function propagateDescMatch(sourceId, changedField) {
  const source = rows.find(r => r.id === sourceId);
  if (!source) return;
  const descEl = document.querySelector(`#tr_${sourceId} td:nth-child(7) input`);
  const desc = (descEl ? descEl.value : source.desc).trim().toLowerCase();
  if (!desc) return;

  rows.forEach(r => {
    if (r.id === sourceId) return;
    // Don't override if user has already manually broken this row free
    if (propagatedRows.has(`manual_${r.id}`)) return;
    const rDescEl = document.querySelector(`#tr_${r.id} td:nth-child(7) input`);
    const rDesc = (rDescEl ? rDescEl.value : r.desc).trim().toLowerCase();
    if (rDesc !== desc) return;

    // Apply the change
    r.acc = source.acc;
    r.vat = source.vat;
    propagatedRows.add(`propagated_${r.id}`);
    const tr = document.getElementById('tr_'+r.id);
    if (tr) {
      tr.innerHTML = buildRowHTML(r);
      if (!r.acc || !r.vat) tr.classList.add('needs-attention');
      else tr.classList.remove('needs-attention');
      // Add undo pill
      showUndoPill(r.id, source.acc, source.vat);
    }
  });
  checkHomeRows();
}

function showUndoPill(id, prevAcc, prevVat) {
  const tr = document.getElementById('tr_'+id);
  if (!tr) return;
  // Inject a small undo notice into the last cell
  const lastTd = tr.querySelector('td:last-child');
  if (!lastTd) return;
  const pill = document.createElement('div');
  pill.id = 'undo_'+id;
  pill.style.cssText = 'font-size:10px;color:var(--blue);cursor:pointer;margin-top:3px;white-space:nowrap;text-decoration:underline;';
  pill.textContent = 'undo';
  pill.title = 'Revert this row — keep editing independently';
  pill.onclick = () => undoPropagate(id, prevAcc, prevVat);
  lastTd.appendChild(pill);
  // Auto-remove after 8 seconds
  setTimeout(() => { pill.remove(); propagatedRows.delete(`propagated_${id}`); }, 8000);
}

function undoPropagate(id, prevAcc, prevVat) {
  const row = rows.find(r => r.id === id);
  if (!row) return;
  row.acc = prevAcc;
  row.vat = prevVat;
  // Mark as manually broken so future propagations skip it
  propagatedRows.add(`manual_${id}`);
  propagatedRows.delete(`propagated_${id}`);
  const tr = document.getElementById('tr_'+id);
  if (tr) {
    tr.innerHTML = buildRowHTML(row);
    if (!row.acc || !row.vat) tr.classList.add('needs-attention');
    else tr.classList.remove('needs-attention');
  }
  checkHomeRows();
  updateSummary();
}

function togglePost(id) {
  const row = rows.find(r => r.id === id);
  if (!row) return;
  row.post = row.post === 'YES' ? 'NO' : 'YES';
  const btn = document.getElementById('postBtn_'+id);
  const tr = document.getElementById('tr_'+id);
  btn.textContent = row.post;
  btn.className = 'toggle-post ' + (row.post==='YES'?'yes':'no');
  tr.classList.toggle('excluded', row.post==='NO');
  updateSummary();
  refreshExportBtn();
}

function toggleHome(id) {
  const row = rows.find(r => r.id === id);
  if (!row) return;
  row.home = !row.home;
  const btn = document.getElementById('homeBtn_'+id);
  btn.className = 'use-home-btn' + (row.home?' active':'');
  btn.textContent = row.home ? '⌂ ON' : '⌂';
  updateNet(id);
  updateSummary();
}

function isHome213(acc) {
  return (acc||'').startsWith('213');
}

function calcNet(gross, bu, home, acc) {
  let amt = parseFloat(gross) || 0;
  if (isHome213(acc)) {
    amt = amt * (getHomePct() / 100);
  } else {
    if (bu !== '' && bu !== null) {
      const buPct = parseFloat(bu);
      if (!isNaN(buPct)) amt = amt * (buPct / 100);
    }
    if (home) {
      amt = amt * (getHomePct() / 100);
    }
  }
  return amt;
}

function updateNet(id) {
  const row = rows.find(r => r.id === id);
  if (!row) return;
  // Read live values from DOM
  const grossEl = document.querySelector(`#tr_${id} input[type=number]`);
  if (grossEl) row.gross = grossEl.value;
  const buEl = document.querySelector(`#tr_${id} .bu-cell input`);
  if (buEl) row.bu = buEl.value;

  const gross = parseFloat(row.gross) || 0;
  const net = calcNet(gross, row.bu, row.home, row.acc);
  const netAdj = (isHome213(row.acc) || row.bu !== '' || row.home) && net !== gross;
  const el = document.getElementById('net_'+id);
  if (el) {
    el.textContent = isNaN(net) ? '—' : `£${net.toFixed(2)}`;
    el.className = 'net-amt' + (netAdj?' adjusted':'');
  }
  updateSummary();
}

  homeMethod = method;

  document.getElementById('hcRoomsInputs').style.display = method==='rooms' ? 'grid' : 'none';

function getHomePct() {
  if (homeMethod === 'pct') {
    const homePct = parseFloat(document.getElementById('hcHomePct').value) || 0;
    const timePct = parseFloat(document.getElementById('hcTimePctA').value) || 0;
    return (homePct / 100) * timePct;
  } else {
    const work  = parseFloat(document.getElementById('hcWorkRooms').value) || 1;
    const total = parseFloat(document.getElementById('hcRooms').value) || 1;
    const timePct = parseFloat(document.getElementById('hcTimePctB').value) || 0;
    return (work / total) * timePct;
  }
}

function calcHome() {
  const pct = getHomePct();

  if (homeMethod === 'pct') {
    const homePct = parseFloat(document.getElementById('hcHomePct').value) || 0;
    const timePct = parseFloat(document.getElementById('hcTimePctA').value) || 0;
    document.getElementById('homePctDisplay').textContent = pct.toFixed(2);
    document.getElementById('hcEffectiveLine').textContent =
      `Effective adjustment: ${homePct}% × ${timePct}% = ${pct.toFixed(2)}% of gross amount`;
  } else {
    const work  = parseFloat(document.getElementById('hcWorkRooms').value) || 1;
    const total = parseFloat(document.getElementById('hcRooms').value) || 1;
    const timePct = parseFloat(document.getElementById('hcTimePctB').value) || 0;
    document.getElementById('homePctDisplayRooms').textContent = pct.toFixed(2);
    document.getElementById('hcEffectiveLine').textContent =
      `Effective adjustment: ${work}/${total} × ${timePct}% = ${pct.toFixed(2)}% of gross amount`;
  }

  // Refresh all 213 rows
  rows.forEach(r => { if (isHome213(r.acc)) updateNet(r.id); });
  updateSummary();
}

// ── SORTING ───────────────────────────────────────────────────────────────────

let sortCol = null;
let sortDir = 1; // 1 = asc, -1 = desc

function getRowValue(row, col) {
  if (col === 'gross') return parseFloat(row.gross) || 0;
  if (col === 'net') {
    const buEl = document.querySelector(`#tr_${row.id} .bu-cell input`);
    const bu = buEl ? buEl.value : row.bu;
    return calcNet(parseFloat(row.gross)||0, bu, row.home, row.acc);
  }
  if (col === 'date') return row.date || '';
  return String(row[col] || '').toLowerCase();
}

function sortBy(col) {
  if (sortCol === col) {
    sortDir *= -1;
  } else {
    sortCol = col;
    sortDir = 1;
  }

  // Update header indicators
  document.querySelectorAll('.sort-ind').forEach(el => el.textContent = '');
  document.querySelectorAll('thead th').forEach(el => el.classList.remove('sort-asc','sort-desc'));
  const indEl = document.getElementById('si-' + col);
  const thEl  = document.querySelector(`thead th[data-col="${col}"]`);
  if (indEl) indEl.textContent = sortDir === 1 ? '▲' : '▼';
  if (thEl)  thEl.classList.add(sortDir === 1 ? 'sort-asc' : 'sort-desc');

  rows.sort((a, b) => {
    const av = getRowValue(a, col);
    const bv = getRowValue(b, col);
    if (av < bv) return -1 * sortDir;
    if (av > bv) return  1 * sortDir;
    return 0;
  });

  // Re-render tbody in new order
  const tbody = document.getElementById('expBody');
  tbody.innerHTML = '';
  rows.forEach(r => renderRow(r));
}

function updateSummary() {
  let posted=0, excl=0, gross=0, net=0;
  rows.forEach(r => {
    if (r.post==='NO') { excl++; return; }
    posted++;
    const g = parseFloat(r.gross)||0;
    gross += g;
    // Read live bu from DOM
    const buEl = document.querySelector(`#tr_${r.id} .bu-cell input`);
    const bu = buEl ? buEl.value : r.bu;
    net += calcNet(g, bu, r.home, r.acc);
  });
  document.getElementById('statPost').textContent = posted;
  document.getElementById('statExcl').textContent = excl;
  document.getElementById('statGross').textContent = '£'+gross.toFixed(2);
  document.getElementById('statNet').textContent = '£'+net.toFixed(2);
}

function fmtDate(dateStr) {
  // Convert YYYY-MM-DD to DD/MM/YYYY for Xero
  if (!dateStr) return '';
  const m = dateStr.match(/^(\d{4})-(\d{2})-(\d{2})$/);
  if (m) return `${m[3]}/${m[2]}/${m[1]}`;
  return dateStr;
}

function exportCSV() {
  const client = document.getElementById('clientName').value.trim() || 'Client';
  const prefix = getPrefix() || 'EXP';
  const defDate = '';
  const defDue  = '';

  const headers = ['*ContactName','*InvoiceNumber','*InvoiceDate','*DueDate',
                   '*Description','*Quantity','*UnitAmount','*AccountCode','*TaxType','TaxAmount'];
  const lines = [headers.join(',')];

  let seq = 1;
  rows.forEach(r => {
    if (r.post !== 'YES') return;

    // Read live values from DOM
    const grossEl = document.querySelector(`#tr_${r.id} input[type=number]`);
    const buEl    = document.querySelector(`#tr_${r.id} .bu-cell input`);
    const descEl  = document.querySelector(`#tr_${r.id} td:nth-child(7) input`);
    const supEl   = document.querySelector(`#tr_${r.id} td:nth-child(6) input`);
    const dateEl  = document.querySelector(`#tr_${r.id} input[type=date]`);
    const refEl   = document.querySelector(`#tr_${r.id} td:nth-child(4) input`);
    const accEl   = document.querySelector(`#tr_${r.id} td:nth-child(2) select`);
    const vatEl   = document.querySelector(`#tr_${r.id} td:nth-child(3) select`);

    const gross  = parseFloat(grossEl?.value || r.gross) || 0;
    const bu     = buEl?.value !== '' ? buEl?.value : r.bu;
    const accFull= accEl?.value || r.acc;
    const accCode= (accFull||'').split(' - ')[0];
    const net    = calcNet(gross, bu, r.home, accFull);
    const desc   = descEl?.value || r.desc;
    const sup    = supEl?.value  || r.supplier || client;
    const dateStr= fmtDate(dateEl?.value || r.date) || defDate;
    const dueStr = defDue || dateStr;
    const ref    = refEl?.value || r.ref || `${prefix}-${seq}`;
    const vat    = vatEl?.value || r.vat;

    // Build description with BU annotation
    let descOut = desc;
    if (isHome213(accFull)) {
      descOut += ` @ ${getHomePct().toFixed(2)}% business use (home)`;
    } else {
      if (bu !== '' && bu !== null && parseFloat(bu) !== 100) {
        descOut += ` @ ${parseFloat(bu).toFixed(0)}% business use`;
      }
      if (r.home) {
        descOut += ` @ ${getHomePct().toFixed(2)}% home office`;
      }
    }

    const row = [
      csvEsc(sup),
      csvEsc(ref),
      dateStr,
      dueStr,
      csvEsc(descOut),
      '1',
      net.toFixed(2),
      accCode,
      csvEsc(vat),
      '',
    ];
    lines.push(row.join(','));
    seq++;
  });

  const blob = new Blob([lines.join('\r\n')], {type:'text/csv'});
  const a = document.createElement('a');
  a.href = URL.createObjectURL(blob);
  const fname = `xero_expenses_${new Date().toISOString().slice(0,10)}.csv`;
  a.download = fname;
  a.click();
  setAlert(`✓ Exported ${seq-1} row${seq-1!==1?'s':''} to ${fname}`, 'success');
}

function csvEsc(val) {
  const s = String(val||'');
  return s.includes(',') || s.includes('"') || s.includes('\n')
    ? `"${s.replace(/"/g,'""')}"` : s;
}
